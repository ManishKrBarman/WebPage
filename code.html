Q1. Find the area of a triangle using Heron's formula

#include <stdio.h>
#include <math.h>
int main(){
int a, b, c;
float s, area;
printf("Enter value of 1st side: ");
scanf("%d", &a);
printf("Enter value of 2nd side: ");
scanf("%d", &b);
printf("Enter value of 3rd side: ");
scanf("%d", &c);
s = (a + b + c) / 2;
area = sqrt(s * (s - a) * (s - b) * (s - c));
printf("Area of the triangle of sides %d, %d & %d is: %f", a, b, c, area);
return 0;
}

Q2. Find the factorial of a number

#include <stdio.h>
int main() {
int n, i, factorial = 1;
printf("Enter a number: ");
scanf("%d", &n);
for (i = 1; i <= n; i++) {
factorial *= i;
}
printf("Factorial of %d is: %d", n, factorial);
return 0;
}

Q3. Find the sum of digits of a number

#include <stdio.h>
int main() {
int n, sum = 0;
printf("Enter a number: ");
scanf("%d", &n);
while (n != 0) {
sum += n % 10;
n = n / 10;
}
printf("Sum of digits of %d is: %d", n, sum);
return 0;
}

Q4. Find the reverse of a number

#include <stdio.h>
int main() {
int n, reverse = 0;
printf("Enter a number: ");
scanf("%d", &n);
while (n != 0) {
reverse = reverse * 10 + n % 10;
n = n / 10;
}
printf("Reverse of %d is: %d", n, reverse);
return 0;
}

Q5. Find the sum of even digits of a number

#include <stdio.h>
int main() {
int n, sum = 0;
printf("Enter a number: ");
scanf("%d", &n);
while (n != 0) {
if ((n % 10) % 2 == 0) {
sum += n % 10;
}
n = n / 10;
}
printf("Sum of even digits of %d is: %d", n, sum);
return 0;
}
